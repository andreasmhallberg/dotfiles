% For in markdownfiles. Use `<text>` to delimit Arabic text.
% Use `\Arabic` `\endArabic` for larger spans of Arabic text. This accepts markdown syntax within. 

\usepackage{polyglossia}
\setotherlanguage{arabic}

\newfontfamily\arabicfont[Script=Arabic, Scale=1.4]{Lateef}
\newfontfamily\A[Script=Arabic, Scale=1.4]{Lateef} % for beamer

% Also in beamer.
% https://tex.stackexchange.com/a/97/44517
\makeatletter
\@ifclassloaded{beamer}
  {% true
    \renewcommand{\texttt}[1]{{\RL{\A#1}}} 
  }
  {% false. for article etc
    \renewcommand{\texttt}[1]{\textarabic{#1}}
  }
\makeatother
